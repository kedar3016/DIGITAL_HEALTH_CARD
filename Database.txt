CREATE TABLE Patient (
    PatientId INT AUTO_INCREMENT PRIMARY KEY,

    Name VARCHAR(50),
    MiddleName VARCHAR(50),
    LastName VARCHAR(50),

    DateOfBirth DATE,
    Age INT,

    BloodGroup VARCHAR(5),

    Email VARCHAR(100),
    PhoneNumber VARCHAR(15),

    Address VARCHAR(255)
);
